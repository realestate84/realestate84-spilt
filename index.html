const functions = require("firebase-functions"); const { google } = require("googleapis"); const fetch = require("node-fetch"); // 서비스 계정 키 — Firebase 환경변수로 관리 (아래 설정 방법 참고) const SERVICE_ACCOUNT = JSON.parse(process.env.DRIVE_SERVICE_ACCOUNT || "{}"); // Google Drive 클라이언트 초기화 function getDriveClient() { const auth = new google.auth.GoogleAuth({ credentials: SERVICE_ACCOUNT, scopes: ["https://www.googleapis.com/auth/drive"], }); return google.drive({ version: "v3", auth }); } /** * Firebase Function: drivePreview * - Firebase Storage URL을 받아 Drive에 업로드 후 임베드 URL 반환 * - 업로드된 파일은 1시간 후 자동 삭제 (cleanup 함께 실행) */ exports.drivePreview = functions .region("asia-northeast3") // 서울 리전 .runWith({ timeoutSeconds: 60, memory: "512MB" }) .https.onCall(async (data, context) => { // 인증된 사용자만 허용 if (!context.auth) { throw new functions.https.HttpsError("unauthenticated", "로그인이 필요합니다."); } const { fileUrl, fileName } = data; if (!fileUrl || !fileName) { throw new functions.https.HttpsError("invalid-argument", "fileUrl, fileName이 필요합니다."); } const ext = fileName.split(".").pop().toLowerCase(); const allowedExts = ["xlsx", "xls", "csv", "docx", "doc"]; if (!allowedExts.includes(ext)) { throw new functions.https.HttpsError("invalid-argument", "지원하지 않는 파일 형식입니다."); } try { // 1. Firebase Storage에서 파일 다운로드 const response = await fetch(fileUrl); if (!response.ok) throw new Error(`파일 다운로드 실패: ${response.status}`); const buffer = await response.buffer(); // 2. MIME 타입 결정 const mimeMap = { xlsx: "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet", xls: "application/vnd.ms-excel", csv: "text/csv", docx: "application/vnd.openxmlformats-officedocument.wordprocessingml.document", doc: "application/msword", }; const sourceMime = mimeMap[ext] || "application/octet-stream"; // Google Drive 변환 MIME (Sheets/Docs로 변환) const convertMime = ["xlsx", "xls", "csv"].includes(ext) ? "application/vnd.google-apps.spreadsheet" : "application/vnd.google-apps.document"; // 3. Drive에 업로드 (Google Sheets/Docs로 변환) const drive = getDriveClient(); const { Readable } = require("stream"); const stream = new Readable(); stream.push(buffer); stream.push(null); const uploadRes = await drive.files.create({ requestBody: { name: fileName, mimeType: convertMime, // 24시간 후 자동 삭제를 위한 description 태그 description: `preview_temp:${Date.now()}`, }, media: { mimeType: sourceMime, body: stream, }, fields: "id,webViewLink", }); const fileId = uploadRes.data.id; // 4. 공개 읽기 권한 부여 (임베드용) await drive.permissions.create({ fileId, requestBody: { role: "reader", type: "anyone", }, }); // 5. 임베드 URL 생성 const isSheet = ["xlsx", "xls", "csv"].includes(ext); const embedUrl = isSheet ? `https://docs.google.com/spreadsheets/d/${fileId}/preview` : `https://docs.google.com/document/d/${fileId}/preview`; return { embedUrl, fileId }; } catch (err) { console.error("drivePreview error:", err); throw new functions.https.HttpsError("internal", err.message); } }); /** * Firebase Function: driveCleanup * - 매일 자정 실행, 24시간 넘은 임시 파일 삭제 */ exports.driveCleanup = functions .region("asia-northeast3") .pubsub.schedule("every 24 hours") .onRun(async () => { const drive = getDriveClient(); const oneDayAgo = new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString(); const res = await drive.files.list({ q: `description contains 'preview_temp:' and createdTime < '${oneDayAgo}'`, fields: "files(id,name)", }); const files = res.data.files || []; console.log(`정리 대상 파일: ${files.length}개`); await Promise.all( files.map((f) => drive.files.delete({ fileId: f.id }).catch((e) => console.warn(`삭제 실패 ${f.id}:`, e.message) ) ) ); return null; });
